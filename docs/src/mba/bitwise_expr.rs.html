1//! This module contains bitwise expressions ([`BExpr`]), which are just bitwise expressions
2//! on n bits, and linear combinations of bitwise expressions ([`LBExpr`]).

9/// LBExpr is short for "Linear combination of Bitwise Expressions"
10/// These are the expressions for which rewrite rules can be efficiently
11/// generated.
12#[derive(Clone, Debug)]
13pub struct LBExpr<R: Ring>(pub Vec<(R::Element, BExpr)>);
14
15impl<R: Ring> LBExpr<R> {
16    /// Create the empty linear combination.
17    /// It evaluates to 0.
18    pub fn zero() -> Self {
19        Self(Vec::new())
20    }
21
22    /// Creates an expression that equals a constant.
23    pub fn constant(c: R::Element, r: &R) -> Self {
24        Self(vec![(r.neg(c), BExpr::Ones)])
25    }
26
27    /// Creates an expression that equals a variable.
28    pub fn var<T: Into<Symbol>>(name: T) -> Self {
29        Self(vec![(R::one(), BExpr::Var(name.into()))])
30    }
31
32    /// Removes all terms with coefficient 0.
33    pub fn remove_zero_terms(&mut self) {
34        self.0.retain(|(i, _)| !i.is_zero());
35    }
36
37    /// Returns all variables in the expression.
38    /// This will include duplicates.
39    pub fn vars(&self) -> Vec<Symbol> {
40        let mut v = BTreeSet::new();
41        self.vars_impl(&mut v);
42        v.into_iter().collect()
43    }
44
45    pub(crate) fn vars_impl(&self, v: &mut BTreeSet<Symbol>) {
46        for (_, e) in &self.0 {
47            e.vars_impl(v);
48        }
49    }
50
51    /// Evaluate an expression with a valuation for the occurring variables.
52    pub fn eval(&self, v: &mut Valuation<R>, r: &R) -> R::Element
53    where
54        R: BinaryRing
55    {
56        self.0.iter()
57            .map(|(i, e)| r.mul(e.eval(v, r), i))
58            .fold(R::zero(), |acc, x| r.add(acc, &x))
59    }
60
61    /// Returns a measure of the complexity of the expression.
62    pub fn complexity(&self) -> u32
63    where
64        R: BinaryRing
65    {
66        // Complexity of a coefficient.
67        let coeff_complexity = |i: &R::Element| {
68            R::count_ones(i) / 2 + R::min_bits(i)
69        };
70
71        self.0.iter()
72            .filter(|(c, _)| !c.is_zero())
73            .map(|(c, e)| coeff_complexity(c) + e.complexity())
74            .sum()
75    }

77    /// Parse a string to an expression.
78    /// Note that this function is extremely limited
79    /// and expects very specific syntax.
80    /// It is used for convenience when testing things and
81    /// not really meant to be used by something outside this crate.
82    pub fn from_string(mut s: String, r: &R) -> Result<Self, String> {
83        s.retain(|c| !matches!(c, ' ' | '\t' | '\n'));
84        let mut it = s.chars().peekable();
85
86        // This stores the current linear combination.
87        let mut v = Vec::new();
88
89        let mut neg = false;
90
91        // Loop over the string/the summands.
92        loop {
93            // Are there still characters left?
94            // If not then we're done.
95            let mut c = match it.peek() {
96                None => return Ok(Self(v)),
97                Some(c) => *c,
98            };
99
100            if c == '-' {
101                neg = !neg;
102                it.next();
103                c = *it.peek().ok_or("Unexpected end of input")?;
104            }
105
106            // If this is a digit then we expect num*BExpr.
107            if c.is_ascii_digit() {
108                // Parse the number.
109                let mut num = r.parse_element(&mut it)
110                    .ok_or("Expected number")?;
111
112                // If the number is negative then negate it.
113                if neg {
114                    num = r.neg(num);
115                }
116
117                // Is it the expected '*'?
118                let lookahead = it.peek();
119                match lookahead {
120                    Some('*') => {
121                        it.next();
122
123                        // Parse the BExpr.
124                        let e = BExpr::parse(&mut it, 0)?;
125
126                        // Push it.
127                        v.push((num, e));
128                    },
129
130                    // If this is a different character then we push -num*(-1).
131                    _ => v.push((r.neg(num), BExpr::Ones)),
132                }
133            } else {
134                // We don't have a factor so just parse the BExpr.
135                let e = BExpr::parse(&mut it, 0)?;
136
137                let sign = match neg {
138                    false => R::one(),
139                    true => r.negative_one(),
140                };
141
142                // Push sign*e.
143                v.push((sign, e));
144            }
145
146            // If the next character is not a plus or - then we are done.
147            match it.peek() {
148                Some('+') => { neg = false; it.next() },
149                Some('-') => { neg = true; it.next() },
150                _ => return Ok(Self(v)),
151            };
152        }
153    }

155    /// Converts an [`LBExpr`] to an [`ExprOp`].
156    pub fn to_expr(&self, r: &R) -> ExprOp<R> {
157        let mut it = self.0.iter();
158        let s = match it.next() {
159            None => return ExprOp::Const(R::zero()),
160            Some(s) => s,
161        };
162
163        let from_summand = |(i, e): &(R::Element, BExpr)| {
164            if let BExpr::Ones = e {
165                ExprOp::Const(r.neg(i.clone()))
166            } else {
167                ExprOp::Mul(ExprOp::Const(i.clone()).into(), e.to_expr(r).into())
168            }
169        };
170
171        let mut cur = from_summand(s);
172        for s in it {
173            cur = ExprOp::Add(cur.into(), from_summand(s).into());
174        }
175
176        cur
177    }
178}

192/// Represents an expression whose value bit `i` depends only on the values of
193/// bits `i` of the variables.
194///
195/// Note that the variant 'Ones' does not equal 1, but a 1 in every bit,
196/// which is -1 in two's complement.
197#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
198pub enum BExpr {
199    Ones,
200    Var(Symbol),
201    And(Box<Self>, Box<Self>),
202    Or(Box<Self>, Box<Self>),
203    Xor(Box<Self>, Box<Self>),
204    Not(Box<Self>),
205}
206
207impl BExpr {
208    pub fn var<T: Into<Symbol>>(name: T) -> Self {
209        Self::Var(name.into())
210    }
211
212    pub fn and(l: Self, r: Self) -> Self {
213        Self::And(l.into(), r.into())
214    }
215
216    pub fn or(l: Self, r: Self) -> Self {
217        Self::Or(l.into(), r.into())
218    }
219
220    pub fn xor(l: Self, r: Self) -> Self {
221        Self::Xor(l.into(), r.into())
222    }
223
224    #[allow(clippy::should_implement_trait)]
225    pub fn not(e: Self) -> Self {
226        Self::Not(e.into())
227    }

229    /// Is the topmost operator a unary operator?
230    /// If there is no operator (i.e. it is [`BExpr::Ones`] or [`BExpr::Var`])
231    /// then this return `true`.
232    pub fn is_topmost_unary(&self) -> bool {
233        matches!(self, Self::Ones | Self::Var(_) | Self::Not(_))
234    }

244    pub(crate) fn vars_impl(&self, v: &mut BTreeSet<Symbol>) {
245        use BExpr::*;
246        match self {
247            Ones => {},
248            Var(c) => drop(v.insert(*c)),
249            And(e1, e2) => { e1.vars_impl(v); e2.vars_impl(v) },
250            Or(e1, e2) => { e1.vars_impl(v); e2.vars_impl(v) },
251            Xor(e1, e2) => { e1.vars_impl(v); e2.vars_impl(v) },
252            Not(e) => e.vars_impl(v),
253        }
254    }

256    /// Evaluate an expression with a valuation for the occurring variables.
257    pub fn eval<R: BinaryRing>(&self, v: &mut Valuation<R>, r: &R) -> R::Element {
258        use BExpr::*;
259        match self {
260            Ones => r.negative_one(),
261            Var(c) => v.value(*c, r).clone(),
262            And(e1, e2) => R::and(e1.eval(v, r), &e2.eval(v, r)),
263            Or(e1, e2) => R::or(e1.eval(v, r), &e2.eval(v, r)),
264            Xor(e1, e2) => R::xor(e1.eval(v, r), &e2.eval(v, r)),
265            Not(e) => r.not(e.eval(v, r)),
266        }
267    }

269    /// Rename a variable.
270    pub fn rename_var(&mut self, old: Symbol, new: Symbol) {
271        use BExpr::*;
272        match self {
273            Ones => (),
274            Var(v) => if *v == old { *v = new },
275            And(l, r) => { l.rename_var(old, new); r.rename_var(old, new) },
276            Or(l, r) => { l.rename_var(old, new); r.rename_var(old, new) },
277            Xor(l, r) => { l.rename_var(old, new); r.rename_var(old, new) },
278            Not(e) => e.rename_var(old, new),
279        }
280    }

282    /// Returns some sort of complexity measure of the expression.
283    pub fn complexity(&self) -> u32 {
284        use BExpr::*;
285        match self {
286            Ones => 1,
287            Var(_) => 1,
288            And(l, r) => l.complexity() + r.complexity() + 1,
289            Or(l, r) => l.complexity() + r.complexity() + 1,
290            Xor(l, r) => l.complexity() + r.complexity() + 1,
291            Not(e) => e.complexity() + 1,
292        }
293    }

298        let mut it = s.chars().peekable();
299
300        Self::parse(&mut it, 0)
301    }

303    pub(self) fn parse(
304        it: &mut std::iter::Peekable<std::str::Chars>,
305        pre: usize
306    ) -> Result<Self, String> {
307        use BExpr::*;
308
309        let c = *it.peek().ok_or("Unexpected end of input")?;
310
311        let mut e = if c == '(' {
312            it.next();
313            let e = Self::parse(it, 0)?;
314            match it.next() {
315                Some(')') => e,
316                _ => return Err("Expected closing parenthesis".to_string()),
317            }
318        } else if c == '~' || c == '!' {
319            it.next();
320            let e = Self::parse(it, 15)?;
321            Not(Box::new(e))
322        } else if c.is_alphabetic() {
323            it.next();
324            let mut var = String::from(c);
325            loop {
326                let Some(c) = it.peek() else {
327                    break
328                };
329
330                if !c.is_alphanumeric() {
331                    break
332                }
333
334                var.push(*c);
335                it.next();
336            }
337
338            Var(var.into())
339        } else if c == '1' {
340            it.next();
341            Ones
342        } else {
343            return Err("Unexpected character".to_string());
344        };
345
346        loop {
347            let c = match it.peek() {
348                None => return Ok(e),
349                Some(c) => *c,
350            };
351
352            let op_pre = match c {
353                '|' => 1,
354                '^' => 2,
355                '&' => 3,
356                _ => return Ok(e),
357            };
358
359            if op_pre <= pre {
360                return Ok(e);
361            }
362
363            // If the current operators precedence is higher than
364            // the one whose subexpressions we are currently parsing
365            // then we need to finish this operator first.
366            it.next();
367            let rhs = Box::new(Self::parse(it, op_pre)?);
368            let lhs = Box::new(e);
369            e = match c {
370                '&' => And(lhs, rhs),
371                '|' => Or(lhs, rhs),
372                '^' => Xor(lhs, rhs),
373                _ => return Err("Unexpected operator".to_string()),
374            };
375        }
376    }

378    /// Converts a [`BExpr`] to an [`ExprOp`].
379    pub fn to_expr<R: Ring>(&self, ring: &R) -> ExprOp<R> {
380        use BExpr::*;
381        match self {
382            Ones => ExprOp::Const(ring.negative_one()),
383            Var(c) => ExprOp::Var(*c),
384            And(l, r) => ExprOp::And(
385                l.to_expr(ring).into(),
386                r.to_expr(ring).into()
387            ),
388            Or(l, r) => ExprOp::Or(
389                l.to_expr(ring).into(),
390                r.to_expr(ring).into()
391            ),
392            Xor(l, r) => ExprOp::Xor(
393                l.to_expr(ring).into(),
394                r.to_expr(ring).into()
395            ),
396            Not(e) => ExprOp::Not(e.to_expr(ring).into()),
397        }
398    }
