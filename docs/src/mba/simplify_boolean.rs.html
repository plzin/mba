1//! Simplify boolean expressions using egg.
2//! There are some obvious improvements you could make:
3//! - Given a truth table, find a minimal expression
4//!   using e.g. the Quine-McCluskey algorithm, before handing
5//!   it to egg.
6//! - Use a custom RewriteScheduler that is aware of
7//!   the cost function, i.e. tries cost reducing rewrites
8//!   more aggressively.
9//! - Maybe a more refined cost function.
10//! - More refined rewrite rules.

16/// The way we simplify is by running equality saturation
17/// on the expression. The egraph for the expression would be
18/// prohibitively large, so that full equality saturation
19/// would take too long. egg has a lot of parameters that
20/// change the behavior of when it stops.
21/// See [here](https://docs.rs/egg/0.10.0/egg/struct.Runner.html)
22/// for the documentation of most of these fields.
23/// When equality saturation stops without the actual
24/// saturation being complete, we start again with the
25/// current best expression. If we do not find a better
26/// expression in the next `retry_same_cost` iterations,
27/// we will stop. We will do this for `retry_limit` times
28/// and return the best expression we found.
29pub struct SimplificationConfig {
30    /// The maximum number of times to start again
31    /// with the current best expression.
32    /// This is not an egg parameter.
33    /// The default is 10, though you might want to
34    /// adjust this, depending on the number of variables
35    /// and other settings.
36    retry_limit: usize,
37    /// The number of times to try to find a better
38    /// expression even though the cost hasn't improved.
39    /// If the expression is the exact same, then
40    /// we return it.
41    /// Say we found an expression with cost 12.
42    /// If we find an expression with cost 12 again,
43    /// we will try again for this number of times,
44    /// as long as the expression changes.
45    /// The default is 3.
46    retry_same_cost: usize,
47    /// The maximum number of nodes in the egraph.
48    /// The default is 10'000.
49    node_limit: usize,
50    /// The maximum number of iterations for the runner.
51    /// The default is 30.
52    iter_limit: usize,
53    /// The maximum time to run for.
54    /// The default is 1s.
55    time_limit: std::time::Duration,
56}
57
58impl Default for SimplificationConfig {
59    /// Returns the default configuration.
60    fn default() -> Self {
61        Self {
62            retry_limit: 10,
63            retry_same_cost: 3,
64            node_limit: 10_000,
65            iter_limit: 30,
66            time_limit: std::time::Duration::from_millis(1000),
67        }
68    }
69}

71define_language! {

73    enum BooleanLanguage {

86// Helper function that checks for equality of two EggExprs.
87// This is not optimized for common subexpressions.
88fn bexpr_eq(l: &EggExpr, r: &EggExpr) -> bool {
89    fn bexpr_eq_impl(l: &EggExpr, lid: Id, r: &EggExpr, rid: Id) -> bool {
90        match (&l[lid], &r[rid]) {
91            (BooleanLanguage::Zero, BooleanLanguage::Zero) => true,
92            (BooleanLanguage::One, BooleanLanguage::One) => true,
93            (BooleanLanguage::Not(lid), BooleanLanguage::Not(rid)) =>
94                bexpr_eq_impl(l, *lid, r, *rid),
95            (BooleanLanguage::And([ll, lr]), BooleanLanguage::And([rl, rr])) =>
96                bexpr_eq_impl(l, *ll, r, *rl) && bexpr_eq_impl(l, *lr, r, *rr),
97            (BooleanLanguage::Or([ll, lr]), BooleanLanguage::Or([rl, rr])) =>
98                bexpr_eq_impl(l, *ll, r, *rl) && bexpr_eq_impl(l, *lr, r, *rr),
99            (BooleanLanguage::Xor([ll, lr]), BooleanLanguage::Xor([rl, rr])) =>
100                bexpr_eq_impl(l, *ll, r, *rl) && bexpr_eq_impl(l, *lr, r, *rr),
101            (BooleanLanguage::Symbol(l), BooleanLanguage::Symbol(r)) => l == r,
102            _ => false,
103        }
104    }
105
106    bexpr_eq_impl(
107        l, Id::from(l.as_ref().len() - 1),
108        r, Id::from(r.as_ref().len() - 1)
109    )
110}

117/// Returns a simplified boolean expression for a given truth table.
118/// The truth table is an iterator over the value of the rows.
119/// This function iterates once over the table to determine
120/// whether the DNF or the CNF is smaller and then calls
121/// `simplify_from_truth_table_dnf` or `simplify_from_truth_table_cnf`.
122/// Thus the iterator needs to be cloneable.
123pub fn simplify_from_truth_table<Table>(
124    vars: &[Symbol], table: Table, cfg: &SimplificationConfig
125) -> BExpr
126where
127    Table: Iterator<Item = bool> + Clone,
128{
129    let (size, ones) = table.clone().fold((0, 0),
130        |(size, ones), v| (size + 1usize, ones + v as usize));
131
132    if ones >= size / 2 {
133        simplify_from_truth_table_cnf(
134            vars,
135            table.enumerate().filter(|(_, v)| !*v).map(|(i, _)| i),
136            cfg
137        )
138    } else {
139        simplify_from_truth_table_dnf(
140            vars,
141            table.enumerate().filter(|(_, v)| *v).map(|(i, _)| i),
142            cfg
143        )
144    }
145}

147/// Returns a simplified boolean expression for a given truth table.
148/// The truth table is an iterator over the indices of the rows
149/// that have a value of 1.
150/// This constructs the DNF and then simplifies it using algebra.
151pub fn simplify_from_truth_table_dnf<Table>(
152    vars: &[Symbol], table: Table, cfg: &SimplificationConfig
153) -> BExpr
154where
155    Table: Iterator<Item = usize>,
156{
157    let mut nodes = Vec::new();
158    for v in vars {
159        nodes.push(BooleanLanguage::Symbol(*v));
160        nodes.push(BooleanLanguage::Not(Id::from(nodes.len() - 1)));
161    }
162
163    let nodes = RefCell::new(nodes);
164
165    fn from_1(
166        vars: &[Symbol], i: usize, nodes: &RefCell<Vec<BooleanLanguage>>
167    ) -> usize {
168        let mut iter = (0..vars.len()).map(|j|
169            j * 2 + (1 - ((i >> j) & 1))
170        );
171
172        let first = iter.next().unwrap();
173        iter.fold(first, |acc, x| {
174            let mut nodes = nodes.borrow_mut();
175            nodes.push(BooleanLanguage::And([Id::from(acc), Id::from(x)]));
176            nodes.len() - 1
177        })
178    }
179
180    let mut iter = table.map(|i| from_1(vars, i, &nodes));
181    let Some(first) = iter.next() else {
182        return BExpr::not(BExpr::Ones);
183    };
184    let root = iter.fold(first, |acc, x| {
185        let mut nodes = nodes.borrow_mut();
186        nodes.push(BooleanLanguage::Or([Id::from(acc), Id::from(x)]));
187        nodes.len() - 1
188    });
189
190    // If there is only one variable `a` and the expression is `a`,
191    // then the variable entry will the root. But the vector already
192    // contains `not a`, so we need to remove it, so that `a` is the root.
193    nodes.borrow_mut().truncate(root + 1);
194
195    let dnf = RecExpr::from(nodes.into_inner());
196    egg_to_bexpr(&simplify_bexpr(&dnf, cfg))
197}

251/// Simplify a boolean expression.
252pub fn simplify(e: &BExpr, cfg: &SimplificationConfig) -> BExpr {
253    let e = bexpr_to_egg(e);
254    let best = simplify_bexpr(&e, cfg);
255    egg_to_bexpr(&best)
256}

258/// Simplifies a EggExpr.
259fn simplify_bexpr(e: &EggExpr, cfg: &SimplificationConfig) -> EggExpr {
260    // The current expression with the best cost and its cost.
261    let mut best_cost = usize::MAX;
262    let mut best = e.clone();
263    let mut best_cost_iter = 0usize;
264    for i in 0.. {
265        let runner = Runner::default()
266            .with_iter_limit(cfg.iter_limit)
267            .with_node_limit(cfg.node_limit)
268            .with_time_limit(cfg.time_limit)
269            .with_expr(&best)
270            .run(&*RULES);
271
272        let stop_reason = runner.stop_reason.unwrap();
273
274        let extractor = Extractor::new(&runner.egraph, AstSize);
275
276        let root = runner.roots[0];
277        let (cost, new_best) = extractor.find_best(root);
278
279        assert!(cost <= best_cost, "Cost should not increase");
280
281        if cost == best_cost {
282            if bexpr_eq(&best, &new_best) {
283                return new_best;
284            } else {
285                best_cost_iter += 1;
286            }
287        } else {
288            best_cost_iter = 0;
289        }
290
291        if best_cost_iter >= cfg.retry_same_cost
292            || matches!(stop_reason, StopReason::Saturated)
293            || i + 1 == cfg.retry_limit {
294            return new_best;
295        }
296        best = new_best;
297        best_cost = cost;
298    }
299
300    unreachable!();
301}

303/// Converts a bitwise expression ([`BExpr`]) to a egg boolean expression ([`EggExpr`]).
304fn bexpr_to_egg(e: &BExpr) -> EggExpr {
305    fn bexpr_to_egg_impl(e: &BExpr, nodes: &mut Vec<BooleanLanguage>) -> Id {
306        match e {
307            BExpr::Var(v) => {
308                nodes.push(BooleanLanguage::Symbol(*v));
309            },
310            BExpr::Ones => nodes.push(BooleanLanguage::One),
311            BExpr::Not(e) => {
312                let e = bexpr_to_egg_impl(e, nodes);
313                nodes.push(BooleanLanguage::Not(e));
314            },
315            BExpr::And(l, r) => {
316                let l = bexpr_to_egg_impl(l, nodes);
317                let r = bexpr_to_egg_impl(r, nodes);
318                nodes.push(BooleanLanguage::And([l, r]));
319            },
320            BExpr::Or(l, r) => {
321                let l = bexpr_to_egg_impl(l, nodes);
322                let r = bexpr_to_egg_impl(r, nodes);
323                nodes.push(BooleanLanguage::Or([l, r]));
324            },
325            BExpr::Xor(l, r) => {
326                let l = bexpr_to_egg_impl(l, nodes);
327                let r = bexpr_to_egg_impl(r, nodes);
328                nodes.push(BooleanLanguage::Xor([l, r]));
329            },
330        }
331
332        Id::from(nodes.len() - 1)
333    }
334
335    let mut nodes = Vec::new();
336    bexpr_to_egg_impl(e, &mut nodes);
337    RecExpr::from(nodes)
338}

340/// Converts an egg boolean expression ([`EggExpr`]) into a bitwise expression ([`BExpr`]).
341fn egg_to_bexpr(e: &EggExpr) -> BExpr {
342    fn bexpr_to_bexpr_impl(e: &EggExpr, idx: Id) -> BExpr {
343        match &e[idx] {
344            BooleanLanguage::Zero => BExpr::not(BExpr::Ones),
345            BooleanLanguage::One => BExpr::Ones,
346            BooleanLanguage::Not(i) => BExpr::not(bexpr_to_bexpr_impl(e, *i)),
347            BooleanLanguage::And([i, j]) =>
348                BExpr::and(bexpr_to_bexpr_impl(e, *i), bexpr_to_bexpr_impl(e, *j)),
349            BooleanLanguage::Or([i, j]) =>
350                BExpr::or(bexpr_to_bexpr_impl(e, *i), bexpr_to_bexpr_impl(e, *j)),
351            BooleanLanguage::Xor([i, j]) =>
352                BExpr::xor(bexpr_to_bexpr_impl(e, *i), bexpr_to_bexpr_impl(e, *j)),
353            BooleanLanguage::Symbol(s) => BExpr::var(s.to_string()),
354        }
355    }
356
357    bexpr_to_bexpr_impl(e, Id::from(e.as_ref().len() - 1))
358}

360/// The equivalences.
361/// Some are definitely redundant,
362/// but I think it is better to have more than less.
363fn make_rules() -> Vec<Rewrite<BooleanLanguage, ()>> {
364    let mut rules = vec![
365        rewrite!("and-0"; "(and ?x 0)" => "0"),
366        rewrite!("or-1"; "(or ?x 1)" => "1"),
367        rewrite!("xor-self"; "(xor ?x ?x)" => "0"),
368
369        // Complementation
370        rewrite!("and-not"; "(and ?x (not ?x))" => "0"),
371        rewrite!("or-not"; "(or ?x (not ?x))" => "1"),
372        rewrite!("xor-not"; "(xor ?x (not ?x))" => "1"),
373
374        // Absorption.
375        rewrite!("and-absorb"; "(and ?x (or ?x ?y))" => "?x"),
376        rewrite!("or-absorb"; "(or ?x (and ?x ?y))" => "?x"),
377    ];
378
379    rules.extend(vec![
380        // Associativity
381        rewrite!("and-assoc"; "(and ?x (and ?y ?z))" <=> "(and (and ?x ?y) ?z)"),
382        rewrite!("or-assoc"; "(or ?x (or ?y ?z))" <=> "(or (or ?x ?y) ?z)"),
383        rewrite!("xor-assoc"; "(xor ?x (xor ?y ?z))" <=> "(xor (xor ?x ?y) ?z)"),
384
385        // Commutativity
386        rewrite!("and-comm"; "(and ?x ?y)" <=> "(and ?y ?x)"),
387        rewrite!("or-comm"; "(or ?x ?y)" <=> "(or ?y ?x)"),
388        rewrite!("xor-comm"; "(xor ?x ?y)" <=> "(xor ?y ?x)"),
389
390        rewrite!("and-1"; "(and ?x 1)" <=> "?x"),
391        rewrite!("and-idemp"; "(and ?x ?x)" <=> "?x"),
392
393        rewrite!("or-0"; "(or ?x 0)" <=> "?x"),
394        rewrite!("or-idemp"; "(or ?x ?x)" <=> "?x"),
395
396        rewrite!("xor-0"; "(xor ?x 0)" <=> "?x"),
397        rewrite!("xor-1"; "(xor ?x 1)" <=> "(not ?x)"),
398
399        // Distributivity
400        rewrite!("and-or"; "(and ?x (or ?y ?z))" <=> "(or (and ?x ?y) (and ?x ?z))"),
401        rewrite!("or-and"; "(or ?x (and ?y ?z))" <=> "(and (or ?x ?y) (or ?x ?z))"),
402        rewrite!("and-xor"; "(and ?x (xor ?y ?z))" <=> "(xor (and ?x ?y) (and ?x ?z))"),
403
404        // Double negation
405        rewrite!("not-not"; "(not (not ?x))" <=> "?x"),
406
407        // De Morgan's laws
408        rewrite!("not-and"; "(not (and ?x ?y))" <=> "(or (not ?x) (not ?y))"),
409        rewrite!("not-or"; "(not (or ?x ?y))" <=> "(and (not ?x) (not ?y))"),
410        rewrite!("not-xor"; "(not (xor ?x ?y))" <=> "(xor (not ?x) ?y)"),
411
412        rewrite!("and-xor-1"; "(and ?x (xor ?x ?y))" <=> "(and ?x (not ?y))"),
413        rewrite!("or-xor-1"; "(or ?x (xor ?x ?y))" <=> "(or ?x ?y)"),
414        rewrite!("xor-and-1"; "(xor ?x (and ?x ?y))" <=> "(and ?x (not ?y))"),
415        rewrite!("xor-or-1"; "(xor ?x (or ?x ?y))" <=> "(and (not ?x) ?y)"),
416    ].concat());
417
418    rules
419}

421/// Verifies that all the equivalences are indeed equivalences.
422#[test]
423fn verify_boolean_equivalences() {
424    fn ast_to_bexpr(e: &PatternAst<BooleanLanguage>) -> BExpr {
425        fn ast_to_bexpr_impl(e: &PatternAst<BooleanLanguage>, idx: Id) -> BExpr {
426            let r = match &e[idx] {
427                ENodeOrVar::Var(v) => return BExpr::Var(v.to_string().into()),
428                ENodeOrVar::ENode(e) => e,
429            };
430
431            match r {
432                BooleanLanguage::Zero => BExpr::not(BExpr::Ones),
433                BooleanLanguage::One => BExpr::Ones,
434                BooleanLanguage::Not(i) => BExpr::not(ast_to_bexpr_impl(e, *i)),
435                BooleanLanguage::And([i, j]) =>
436                    BExpr::and(ast_to_bexpr_impl(e, *i), ast_to_bexpr_impl(e, *j)),
437                BooleanLanguage::Or([i, j]) =>
438                    BExpr::or(ast_to_bexpr_impl(e, *i), ast_to_bexpr_impl(e, *j)),
439                BooleanLanguage::Xor([i, j]) =>
440                    BExpr::xor(ast_to_bexpr_impl(e, *i), ast_to_bexpr_impl(e, *j)),
441                BooleanLanguage::Symbol(_) => panic!("There should be no free variables."),
442            }
443        }
444
445        ast_to_bexpr_impl(e, Id::from(e.as_ref().len() - 1))
446    }
447
448    fn verify(lhs: &PatternAst<BooleanLanguage>, rhs: &PatternAst<BooleanLanguage>) {
449        use crate::valuation::Valuation;
450        use crate::rings::U8;
451        let lhs = ast_to_bexpr(lhs);
452        let rhs = ast_to_bexpr(rhs);
453        println!("Verifying equivalence: {lhs} == {rhs}");
454        let mut vars = std::collections::BTreeSet::new();
455        lhs.vars_impl(&mut vars);
456        rhs.vars_impl(&mut vars);
457        let vars: Vec<_> = vars.into_iter().collect();
458        let mut val = Valuation::zero();
459
460        assert!(vars.len() <= 64);
461
462        for i in 0..(1u8 << vars.len()) {
463            for (j, v) in vars.iter().enumerate() {
464                val.set_value(*v, (i >> j) & 1);
465            }
466
467            assert_eq!(lhs.eval(&mut val, &U8), rhs.eval(&mut val, &U8),
468                "Invalid boolean equivalence: {lhs} != {rhs} for {val:?}");
469        }
470    }
471
472    for rule in &*RULES {
473        let Some(lhs) = rule.searcher.get_pattern_ast() else {
474            continue;
475        };
476
477        let Some(rhs) = rule.applier.get_pattern_ast() else {
478            continue;
479        };
480
481        verify(lhs, rhs);
482    }
483}

485fn n_vars(n: usize) -> Vec<Symbol> {
486    if n <= 26 {
487        return (b'a'..=b'z').take(n)
488            .map(|c| Symbol::from(std::str::from_utf8(&[c]).unwrap()))
489            .collect();
490    }
491
492    (1..=n).map(|i| Symbol::from(format!("x{i}"))).collect()
493}

513        &cfg

521fn list_minimal_expressions() {
522    let vars = n_vars(3);
523    let mut truth_table = vec![false; 1 << vars.len()];
524    let cfg = SimplificationConfig::default();
525
526    loop {
527        let e = simplify_from_truth_table(&vars, truth_table.iter().cloned(), &cfg);
528        for v in &truth_table {
529            print!("{}", *v as usize);
530        }
531        println!(": {e}");
532
533        // Is it all ones?
534        if truth_table.iter().all(|v| *v) {
535            break;
536        }
537
538        // Binary addition.
539        for v in &mut truth_table {
540            *v = !*v;
541            if *v {
542                break;
543            }
544        }
