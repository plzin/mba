1//! A trait for the kind of number
2//! ([ring](https://en.wikipedia.org/wiki/Ring_(mathematics))) we are working
3//! with. These traits exist and look the way they do for multiple reasons:
4//!
5//! - So we can have extremely fast implementations for common bit widths,
6//!   i.e. 8, 16, 32, 64, that use single instructions for wrapping operations
7//!   and don't have to go through an insane amount of `BigInt` overhead.
8//! - We can have slower but more general implementations using `BigInt`s
9//!   that allow the modulus to be set at runtime rather than compile time.
10//!
11//! The most important trait is [`Ring`] which stores information about the
12//! ring we are working in and all operations on the ring are implemented in
13//! this trait. E.g. or [`U8`], [`U16`], it is empty, and for [`BigIntModN`]
14//! it contains the modulus.
15//!
16//! An instance of [`Ring`] needs to be passed to any function that uses the
17//! ring which seems unnecessary for [`U8`] but for [`BigIntModN`] we don't
18//! know how to (e.g.) add elements otherwise. The other option would be to
19//! store the modulus with every element of the ring which would waste memory
20//! and we'd have to check that the moduli are equal for arithmetic operations.
21//!
22//! The [`BinaryRing`] trait is a trait that essentially just means
23//! that we are in a ring modulo a power of two.
24//! This allows us in [`BinaryBigInt`] to just store the number of bits instead
25//! of the whole modulus.
26//! It has the bitwise operations on it.

59/// Choose the most efficient [`BinaryRing`] for the given number of bits.
60#[macro_export]
61macro_rules! choose_binary_ring {
62    ($e:expr, $r:ident = $bits:expr) => {{
63        let bits = $bits;
64        if bits < 8 {
65            let $r = $crate::rings::VarU8::new(bits);
66            $e
67        } else if bits == 8 {
68            let $r = $crate::rings::U8;
69            $e
70        } else if bits < 16 {
71            let $r = $crate::rings::VarU16::new(bits);
72            $e
73        } else if bits == 16 {
74            let $r = $crate::rings::U16;
75            $e
76        } else if bits < 32 {
77            let $r = $crate::rings::VarU32::new(bits);
78            $e
79        } else if bits == 32 {
80            let $r = $crate::rings::U32;
81            $e
82        } else if bits < 64 {
83            let $r = $crate::rings::VarU64::new(bits);
84            $e
85        } else if bits == 64 {
86            let $r = $crate::rings::U64;
87            $e
88        } else if bits < 128 {
89            let $r = $crate::rings::VarU128::new(bits);
90            $e
91        } else if bits == 128 {
92            let $r = $crate::rings::U128;
93            $e
94        } else {
95            let $r = $crate::rings::BinaryBigInt::new(bits);
96            $e
97        }
98    }};
99}

101/// This should eventually be the fastest way to initialize a [`BigUint`] with a
102/// power of two. But I haven't actually measured it so I am just guessing that
103/// this is faster than `(BigUint::one() << bit)`.
104pub(crate) fn biguint_pow2(bit: u32) -> BigUint {
105    let mut i = <BigUint as Zero>::zero();
106    i.set_bit(bit as u64, true);
107    i
108}

110/// When the `m` is guaranteed to be in `[0, 2n)` where `n` is the modulus,
111/// reduce mod n. I am guessing checking if `m >= n` and subtracting `n`
112/// will be a lot faster than `%`.
113pub(crate) fn reduce_simple(m: &mut BigUint, n: &BigUint) {
114    if &*m >= n {
115        *m -= n;
116    }
117}

131/// Negates the element mod m without allocating.
132/// See [`neg_assign`].
133pub(crate) fn neg_assign_mod(e: &mut BigUint, m: &BigUint) {
134    if Zero::is_zero(e) {
135        return;
136    }
137    *e = m - std::mem::take(e);

146/// Bitwise negates the element mod m without allocating.
147/// See [`neg_assign`].
148///
149/// The way this works is by expressing the negation using arithmetic.
150/// Usually to subtract e from a, we compute the two's complement of e and add
151/// it to a. The two's complement means negating the bits of e and adding one.
152/// So, `0 - e = -e = !e + 1` => `!e = -e - 1 = -(e + 1)`.
153/// Since we are doing this arithmetic on `BigUint`s, which don't do wrapping
154/// arithmetic, we need to be careful about two things:
155/// - Since `0 <= e < m`, `-m <= -(e + 1) < 0`, so we can add `m` (subtract this
156///   from `m`) to get a value `0 <= m - (e + 1) < m` which is the correct
157///   representative.
158/// - We also need to be careful to never use negative numbers during the
159///   computation, since `BigUint` can only store non-negative integers.
160///   If we wrote this as `-e - 1 + m` we would be using negative numbers, but
161///   by subtracting the `e + 1` from `m` it doesn't.
162pub(crate) fn not_assign_mod(e: &mut BigUint, m: &BigUint) {
163    *e = m - (std::mem::take(e) + 1u32);
164
165    // This is unsound because `sub` can panic when you use the
166    // `Ring` functions with invalid representatives.
167    //unsafe {
168    //    std::ptr::write(e, m - (std::ptr::read(e) + 1u32))
169    //}
170}

178    pub fn test_inverse<R: Ring>(r: &R) {
179        let rng = &mut StdRng::seed_from_u64(0);
180
181        for _ in 0..1000 {
182            let e = r.random(rng);
183            let inv = r.inverse(&e);
184            let is_unit = r.is_unit(&e);
185            if let Some(inv) = inv {
186                let prod = r.mul(e.clone(), &inv);
187                assert_eq!(prod, R::one(), "`inverse` returned {inv} but {e} * {inv} = {prod}");
188                assert!(is_unit, "`inverse` returned {inv} but `is_unit` returned false");
189            } else {
190                assert!(!is_unit, "`inverse` returned None but `is_unit` returned true");
191            }
192        }
193    }

195    pub fn test_element_from_biguint<R>(r: &R)
196    where
197        R: BinaryRing,
198    {
199        let mut rng = StdRng::seed_from_u64(0);
200        for _ in 0..1000 {
201            let n = r.random(&mut rng);
202            let biguint = R::to_representative(&n);
203            let check = r.element_from_biguint(&biguint);
204            assert_eq!(n, check, "element_from_biguint({biguint}) = {n} but got {check}");
205        }
206    }

208    pub fn test_rounded_div<R>(r: &R)
209    where
210        R: BinaryRing,
211        BigInt: From<R::Element>,
212    {
213        let mut rng = StdRng::seed_from_u64(0);
214        for _ in 0..1000 {
215            let a = r.random(&mut rng);
216            let b = loop {
217                let b = r.random(&mut rng);
218                if !b.is_zero() {
219                    break b;
220                }
221            };
222            let q = R::rounded_div(&a, &b);
223            let check = r.element_from_bigint(&BigRational::new(
224                BigInt::from(a.clone()),
225                BigInt::from(b.clone()),
226            ).round().into_raw().0);
227            assert_eq!(q, check,
228                "rounded_div({a}, {b}) = {check} but got {q}");
229        }
230    }
