12/// A "lattice", i.e. the span of a set of vectors.
13/// This can never be empty because it always contains the zero vector even if
14/// [`Lattice::basis`] has zero rows. The matrix always stores the ambient
15/// dimension (the number of columns) even if there are zero rows.
16#[derive(Clone, Debug, PartialEq)]
17pub struct Lattice<R: Ring> {
18    /// The basis matrix of the lattice.
19    /// The basis vectors are the rows of the matrix.
20    pub basis: OwnedMatrix<R>,
21}
22
23impl<R: Ring> Lattice<R> {
24    /// Creates a lattice that contains only the zero vector.
25    pub fn zero(ambient_dim: usize) -> Self {
26        Self {
27            basis: OwnedMatrix::zero(0, ambient_dim)
28        }
29    }
30
31    /// The lattice basis are the rows of the matrix.
32    pub fn from_basis(basis: OwnedMatrix<R>) -> Self {
33        Self { basis }
34    }
35
36    /// The rows of the matrix generate the lattice
37    /// but are potentially linearly dependent.
38    /// This function will compute the Hermite normal form
39    /// and remove zero rows.
40    pub fn from_generating_set(
41        mut generating_set: OwnedMatrix<R>,
42        r: &R,
43    ) -> Self
44    where
45        R: IntDivRing,
46    {
47        hermite_normal_form(&mut generating_set, r);
48        generating_set.remove_zero_rows();
49        Self {
50            basis: generating_set
51        }
52    }
53
54    /// Returns the rank of the lattice, i.e. the number if basis vectors.
55    pub fn rank(&self) -> usize {
56        self.basis.num_rows()
57    }
58
59    /// Returns the dimension of the ambient space or 0 if the lattice is empty.
60    pub fn ambient_dim(&self) -> usize {
61        self.basis.num_cols()
62    }
63
64    /// Returns the vector on the lattice that is the linear combination of the
65    /// basis vectors with the given coefficients.
66    pub fn at<S: VectorStorage<R> + ?Sized>(
67        &self,
68        coefficients: &Vector<R, S>,
69        r: &R,
70    ) -> OwnedVector<R> {
71        self.at_impl(coefficients, OwnedVector::zero(self.ambient_dim()), r)
72    }
73
74    /// Adds the vector on the lattice that is the linear combination of the
75    /// basis vectors with the given `coefficients` to the `initial` vector.
76    pub(self) fn at_impl<S: VectorStorage<R> + ?Sized>(
77        &self,
78        coefficients: &Vector<R, S>,
79        initial: OwnedVector<R>,
80        r: &R,
81    ) -> OwnedVector<R> {
82        assert!(coefficients.dim() == self.rank());
83        self.basis.rows().zip(coefficients).fold(
84            initial,
85            |acc, (e, c)| acc.mul_add(c, e, r)
86        )
87    }

89    /// Returns a random point on the lattice.
90    pub fn sample_point<Rand: Rng>(
91        &self,
92        rng: &mut Rand,
93        r: &R,
94    ) -> OwnedVector<R> {
95        self.sample_point_impl(Vector::zero(self.ambient_dim()), rng, r)
96    }

98    /// Samples a point from the lattice that is added to the initial vector.
99    pub(self) fn sample_point_impl<Rand: Rng>(
100        &self,
101        initial: OwnedVector<R>,
102        rng: &mut Rand,
103        r: &R,
104    ) -> OwnedVector<R> {
105        assert!(initial.dim() == self.ambient_dim());
106
107        self.basis.rows()
108            .fold(initial, |acc, b| acc.mul_add(&r.random(rng), b, r))
109    }

111    /// Returns an integer lattice such that two lattices are equal (i.e.
112    /// contain the same points) iff `l1.canonicalize() == l2.canonicalize()`.
113    ///
114    /// The basis of the returned lattice is guaranteed to be in Hermite normal
115    /// form and of full rank. In particular, the diagonal entries are positive.
116    pub fn canonicalize(&self, ring: &R) -> Lattice<Z>
117    where
118        R: BinaryRing,
119    {
120        // Transform the matrix and vector to integer matrices and vectors.
121        let mut a = self.basis.transform(|e| R::to_representative(e).into());
122
123        // Append rows that contain the modulus.
124        a.append_zero_rows(a.num_cols());
125        for i in 0..a.num_cols() {
126            a[(self.basis.num_rows() + i, i)] = Z::one() << ring.bits();
127        }
128
129        // Compute the HNF of the integer matrix.
130        let _ = hermite_normal_form(&mut a, &Z);
131
132        a.remove_zero_rows();
133
134        Lattice { basis: a }
135    }

140    pub fn size_reduce(&mut self, r: &R)
141    where

143    {
144        size_reduce(&mut self.basis, r);
145    }

217/// A lattice that is offset from the origin by a vector.
218/// Mathematicians would call this a lattice coset.
219///
220/// In contrast to [`Lattice`], this can be empty which is represented by an
221/// offset vector of dimension 0. The [`AffineLattice::lattice`] still stores
222/// the ambient dimension.
223#[derive(Clone, Debug, PartialEq)]
224pub struct AffineLattice<R: Ring> {
225    pub offset: OwnedVector<R>,
226    pub lattice: Lattice<R>,
227}
228
229impl<R: Ring> AffineLattice<R> {
230    /// Creates an empty affine lattice that contains no points.
231    pub fn empty(ambient_dim: usize) -> Self {
232        Self {
233            offset: Vector::empty(),
234            lattice: Lattice::zero(ambient_dim),
235        }
236    }
237
238    /// Creates an affine lattice from an offset and a basis.
239    pub fn from_offset_basis(
240        offset: OwnedVector<R>,
241        basis: OwnedMatrix<R>,
242    ) -> Self {
243        Self {
244            offset,
245            lattice: Lattice::from_basis(basis),
246        }
247    }
248
249    /// The ambient dimension of the lattice.
250    pub fn ambient_dim(&self) -> usize {
251        self.lattice.ambient_dim()
252    }
253
254    /// Is this lattice empty?
255    pub fn is_empty(&self) -> bool {
256        self.offset.is_empty()
257    }
258
259    /// Returns the vector on the lattice that is the linear combination of the
260    /// basis vectors with the given coefficients.
261    pub fn at<S: VectorStorage<R> + ?Sized>(
262        &self,
263        coefficients: &Vector<R, S>,
264        r: &R,
265    ) -> OwnedVector<R> {
266        self.lattice.at_impl(coefficients, self.offset.clone(), r)
267    }
268
269    /// Returns a random point on the lattice.
270    pub fn sample_point<Rand: Rng>(
271        &self,
272        rng: &mut Rand,
273        r: &R,
274    ) -> OwnedVector<R> {
275        self.lattice.sample_point_impl(self.offset.clone(), rng, r)
276    }
277
278    /// Creates a matrix `M` such that affine lattice `L1` and `L2` are equal,
279    /// i.e. contain the same points, iff `canonicalize(L1) = canonicalize(L2)`.
280    pub fn canonicalize(&self, ring: &R) -> AffineLattice<Z>
281    where
282        R: BinaryRing,
283    {
284        if self.is_empty() {
285            return AffineLattice::empty(self.ambient_dim());
286        }
287
288        let lattice = self.lattice.canonicalize(ring);
289        let mut b = self.offset.transform(|e| R::to_representative(e).into());
290
291        // Reduce the entries of the vector.
292        b.reduce(&lattice.basis, &Z);
293
294        AffineLattice { offset: b, lattice }
295    }
296
297    /// Returns the shortest vector on the **affine(!)** lattice.
298    /// This solves the CVP on the normal lattice with
299    /// target offset.
300    pub fn svp<W: WorkingTypeFor<R>>(
301        &self,
302        rad: Option<W::Element>,
303        wt: &W,
304        r: &R,
305    ) -> Option<OwnedVector<R>> {
306        self.lattice.cvp_planes(self.offset.view(), rad, wt, r)
307            .map(|v| self.offset.sub_rhs(v, r))
308    }
309}

311/// Size reduce the basis.
312/// This essentially is Gram-Schmidt
313/// but rounding the coefficients to integers.
314pub fn size_reduce<R: IntDivRing>(basis: &mut OwnedMatrix<R>, r: &R) {
315    for i in 0..basis.num_rows() {
316        for j in 0..i {
317            let (b_i, b_j) = basis.get_rows_mut(i, j);
318            let dot = b_i.dot(b_j, r);
319            let b_j_norm_sqr = b_j.norm_sqr(r);
320            let q = R::rounded_div(&dot, &b_j_norm_sqr);
321
322            if q.is_zero() {
323                continue;
324            }
325
326            let q = r.neg(q);
327            b_i.mul_add_assign(&q, b_j, r);
328        }
329    }
330}

332/// Performs LLL basis reduction using rational numbers.
333pub fn lll<R: IntDivRing, W: WorkingTypeFor<R>>(
334    basis: &mut OwnedMatrix<R>,
335    delta: &W::Element,
336    wt: &W,
337    r: &R,
338) {
339    let n = basis.num_rows();
340    let mut did_swap = true;
341
342    while did_swap {
343        size_reduce(basis, r);
344
345        // Lovasz condition
346        did_swap = false;
347        for i in 0..n-1 {
348            let b = &basis[i];
349            let c = &basis[i+1];
350
351            let lhs = wt.from_ring(&c.norm_sqr(r), r);
352
353            let b_norm_sqr = wt.from_ring(&b.norm_sqr(r), r);
354            let q = wt.div(wt.from_ring(&c.dot(b, r), r), &b_norm_sqr);
355            let rhs = wt.mul(
356                b_norm_sqr,
357                &(wt.add(wt.neg(wt.square(q)), delta))
358            );
359
360            if wt.is_lt(&lhs, &rhs) {
361                basis.swap_rows(i, i + 1);
362                did_swap = true;
363                break;
364            }
365        }
366    }
367}

369/// Approximates the CVP using Babai's rounding technique.
370/// The returns value is a vector of the coefficients
371/// of the linear combination of the basis vectors,
372/// so if you want the actual point,
373/// you still have to matrix multiply with the basis matrix.
374/// In practice, it is a good idea to reduce the basis (e.g. using LLL)
375/// so that the approximation is good.
376pub fn cvp_rounding<R: Ring, W: WorkingTypeFor<R>>(
377    basis: &OwnedMatrix<R>,
378    t: &VectorView<R>,
379    wt: &W,
380    ring: &R,
381) -> OwnedVector<R> {
382    let mut a = OwnedMatrix::zero(basis.num_cols(), basis.num_rows());
383
384    for r in 0..a.num_rows() {
385        for c in 0..a.num_cols() {
386            a[(r, c)] = wt.from_ring(&basis[(c, r)], ring);
387        }
388    }
389
390    let b = OwnedVector::from_iter(
391        t.dim(), t.iter().map(|i| wt.from_ring(i, ring))
392    );
393
394    // If the system has full rank,
395    // then we can just use an exact solution.
396    let x = if t.dim() == basis.num_rows() {
397        solve_linear(a, b, wt)
398    }
399
400    // Otherwise we can treat it as a Ordinary Least Squares problem,
401    // i.e. find the point in the subspace spanned by the vectors
402    // that is closest to the given point.
403    else {
404        let a_t = a.transpose();
405        solve_linear(a_t.mul(&a, wt), a_t.mul_vec_post(&b, wt), wt)
406    };
407
408    let x = x.expect("Basis vectors were not independent.");
409
410    let mut res = OwnedVector::zero(x.dim());
411    for (i, f) in res.iter_mut().zip(x.iter()) {
412        *i = W::to_ring(f, ring);
413    }
414
415    res
416}

418/// Gram-Schmidt of the rows of the matrix `a`.
419/// The rows will span the same subspace as the original rows,
420/// even if they are not linearly independent.
421/// The rows of the result are not normalized.
422/// Call `gram_schmidt_orthonormal` if you want that.
423pub fn gram_schmidt<R: Field>(mut a: OwnedMatrix<R>, r: &R) -> OwnedMatrix<R> {
424    for i in 0..a.num_rows() {
425        for j in 0..i {
426            let (a_i, a_j) = a.get_rows_mut(i, j);
427            let f = r.neg(r.div(a_j.dot(a_i, r), &a_j.norm_sqr(r)));
428            a_i.mul_add_assign(&f, a_j, r);
429        }
430    }
431
432    a
433}

435/// Gram-Schmidt of the rows of the matrix `a`.
436/// The rows of the result are normalized.
437/// Call `gram_schmidt` if you don't want that.
438/// This functions requires that the rows are linearly independent
439/// (unlike `gram_schmidt`).
440pub fn gram_schmidt_orthonormal<R: Field + SqrtRing>(
441    mut a: OwnedMatrix<R>,
442    r: &R,
443) -> OwnedMatrix<R> {
444    a = gram_schmidt(a, r);
445    for i in 0..a.num_rows() {
446        a[i].normalize(r);
447    }
448    a
449}

451/// Computes the RQ-decomposition (row QR-decomposition) of a matrix.
452/// Returns a lower triangular matrix `R` and an orthogonal matrix `Q`,
453/// such that `R * Q = A`.
454pub fn rq_decomposition<R: Field + SqrtRing>(
455    a: &OwnedMatrix<R>,
456    r: &R,
457) -> (OwnedMatrix<R>, OwnedMatrix<R>) {
458    let q = gram_schmidt_orthonormal(a.clone(), r);
459    let r = a.mul(q.transpose(), r);
460    (r, q)
461}

463/// Approximates the CVP using Babai's nearest plane algorithm.
464/// The return value is a pair of the vector and the vector of coefficients
465/// of that vector if `compute_coefficients` is true.
466/// Otherwise the second element of the pair is an empty vector.
467pub fn cvp_nearest_plane<R: Ring, W: WorkingTypeFor<R>>(
468    basis: &OwnedMatrix<R>,
469    t: &VectorView<R>,
470    compute_coefficients: bool,
471    wt: &W,
472    r: &R,
473) -> (OwnedVector<R>, OwnedVector<R>) {
474    let q = gram_schmidt(basis.transform(|i| wt.from_ring(i, r)), wt);
475
476    let mut off: OwnedVector<R> = t.to_owned();
477    let b = off.transform::<W, _>(|i| wt.from_ring(i, r));
478
479    let mut coeff = if compute_coefficients {
480        OwnedVector::zero(basis.num_rows())
481    } else {
482        OwnedVector::empty()
483    };
484
485    for i in (0..basis.num_rows()).rev() {
486        let a = &q[i];
487        // c = round(⟨a, b⟩ / ⟨a, a⟩)
488        let c = b.iter().zip(a.iter())
489            .fold(W::Element::zero(), |acc, (i, f)| wt.mul_add(acc, i, f));
490        let c = wt.div(c, &a.norm_sqr(wt));
491        let c = r.neg(W::to_ring(&c, r));
492        off.mul_add_assign(&c, &basis[i], r);
493        if compute_coefficients {
494            coeff[i] = c;
495        }
496    }
497
498    (t.sub_rhs(off, r), coeff)
499}

501/// Solves the CVP exactly.
502/// - `basis` is the matrix that contains the basis as rows.
503/// - `t` is the target vector.
504/// - `prec` is the precision of the floating point numbers used.
505/// - `rad_sqr` is an optional float that contains the square of the maximum
506///   distance to search for.
507///
508/// The returned vector is the vector of coefficients.
509/// This will always return some vector unless no vector is within `r` of the target.
510/// This algorithm is a generalization Babai's nearest plane algorithm
511/// that searches all planes that could contain the closest vector.
512/// It is the simplest one I could think of.
513pub fn cvp_planes<R: Ring, W: WorkingTypeFor<R>>(
514    basis: &OwnedMatrix<R>,
515    t: &VectorView<R>,
516    rad_sqr: Option<W::Element>,
517    wt: &W,
518    ring: &R,
519) -> Option<OwnedVector<R>> {
520    assert!(basis.num_cols() == t.dim(),
521        "Mismatch of basis/target vector dimension.");
522    let bf = basis.transform::<W, _>(|i| wt.from_ring(i, ring));
523
524    // Q is the Gram-Schmidt orthonormalization and
525    // R is the basis matrix with respect to the Gram-Schmidt basis.
526    let (r, q) = rq_decomposition(&bf, wt);
527    assert!(r.num_cols() == r.num_rows());
528
529    // Write the target vector in that basis.
530    // If the vector is not in the span of the basis,
531    // then this will project it into the span.
532    let qt = q.mul_vec_post(&t.transform(|i| wt.from_ring(i, ring)), wt);
533
534    // We just do this to avoid having to pass around an Option.
535    let rad = rad_sqr.unwrap_or_else(|| wt.infinity());
536
537    // Multiply the coefficients of the vector.
538    return cvp_impl(r.num_cols() - 1, &r, qt.view(), &rad, wt, ring).map(|v| v.0);
539
540    /// This actually finds the closest point.
541    /// `rad` is the squared norm.
542    /// This returns the coordinates of the closest point in the basis
543    /// as the first entry and the actual point as the second entry.
544    fn cvp_impl<R: Ring, W: WorkingTypeFor<R>>(
545        i: usize,
546        r: &OwnedMatrix<W>,
547        qt: &VectorView<W>,
548        rad: &W::Element,
549        wt: &W,
550        ring: &R,
551    ) -> Option<(OwnedVector<R>, OwnedVector<W>)> {
552        // One dimensional lattice.
553        if i == 0 {
554            let qt = &qt[0];
555            let r = &r[(0, 0)];
556            // `m` is the index of the closest point.
557            // `d` is the distance to it.
558            let m = W::round(wt.div(qt.clone(), r));
559            let plane = wt.mul(r.clone(),  &m);
560            let d = wt.square(wt.sub(plane.clone(), qt));
561            return wt.is_le(&d, rad).then(|| (
562                OwnedVector::from_array([W::to_ring(&m, ring)]),
563                OwnedVector::from_array([plane])
564            ));
565        }
566
567        let qtc = &qt[i];
568        let rc = &r[(i, i)];
569
570        // Index of the closest plane before rounding.
571        let start_index_fl = wt.div(qtc.clone(), rc);
572        let start_index = W::round(start_index_fl.clone());
573
574        // Suppose the start index was -0.4.
575        // We would want to check the planes in the order
576        // 0, -1, 1, -2, 2.
577        // But if the start index was 0.4, we would want
578        // 0, 1, -1, 2, -2.
579        // So depending on whether we round up or down
580        // to the integer start index, we will negate
581        // the offset from the start index.
582        let negate_offset = wt.is_lt(&start_index_fl, &start_index);
583
584        // The current plane's offset.
585        let mut offset = 0isize;
586
587        let mut min_dist = rad.clone();
588        let mut min = None;
589
590        // Iterate over all possible planes.
591        loop {
592            // Index of the plane we are considering.
593            let index = W::add_isize(start_index.clone(), offset);
594
595            // Compute the index offset of the next plane.
596            // Negate the offset first.
597            offset = -offset;
598
599            // If we negate the offsets, i.e. 0, -1, 1, -2, ...,
600            // then if we are <= 0, we need to subtract 1.
601            // E.g. if the offset was 1, then we negated it to -1
602            // and subtract 1.
603            if negate_offset && offset <= 0 {
604                offset -= 1;
605            }
606
607            // In the 0, 1, -1, 2, ... case,
608            // if the offset is >= 0, we need to add 1.
609            // E.g. if the offset was -1, then we negated it to 1
610            // and add 1.
611            else if !negate_offset && offset >= 0 {
612                offset += 1;
613            }
614
615            // If this overflows, you probably would have
616            // ctrl-c'd before we got here.
617            // debug_assert!(offset != 0);
618
619            // Distance to the plane.
620            // This is the distance of the target to
621            // its orthogonal projection in the plane.
622            let d = wt.square(wt.sub(wt.mul(index.clone(), rc), qtc));
623
624            // If the plane is not in the radius,
625            // then the next one in the loop definitely is not
626            // either, because of the way we iterate over the planes.
627            if wt.is_gt(&d, &min_dist) {
628                break;
629            }
630
631            // We can use a smaller radius inside the plane.
632            // The target to its projection to any point
633            // in the plane form a right triangle.
634            // So by Pythagoras the squared distance in the
635            // plane can only be <= rad - d.
636            // rad and d are already the square of the distance.
637            let plane_dist = wt.sub(min_dist.clone(), &d);
638
639            // Compute the point in the plane we need to be close to now.
640            let point_in_plane = qt.sub_rhs(r[i].mul_ref(&index, wt), wt);
641
642            // Recursively find the closest point.
643            let Some((mut v, mut w)) = cvp_impl(
644                i - 1, r, point_in_plane.view(), &plane_dist, wt, ring
645            ) else {
646                continue
647            };
648            assert!(v.dim() == i);
649
650            // w is the closest point.
651            // It is the closest point of the previous call,
652            // plus the index of the plane times the basis vector
653            // of the current iteration.
654            w.mul_add_assign(&index, &r[i][0..i], wt);
655            w.append(wt.mul(index.clone(), rc));
656
657            // Compute the distance to the point.
658            let d = w.dist_sqr(&qt[0..i+1], wt);
659
660            // If the distance is smaller than the current minimal dist,
661            // then we have found the new best point.
662            if wt.is_le(&d, &min_dist) {
663                // v is the new coordinate vector of the point.
664                // It is the coordinates of the closest point of
665                // the previous call, concat the index of the plane.
666                v.append(W::to_ring(&index, ring));
667
668                min = Some((v, w));
669                min_dist = d;
670            }
671        }
672
673        min
674    }
675}

677/// Solves a square system of linear equations.
678fn solve_linear<W: WorkingType>(
679    mut a: OwnedMatrix<W>,
680    mut b: OwnedVector<W>,
681    wt: &W
682) -> Option<OwnedVector<W>> {
683    assert!(a.num_rows() == a.num_cols(),
684        "This function only supports non-singular square systems.");
685    let rank = a.num_rows();
686    for i in 0..rank {
687        // Choose a pivot in the c-th column.
688        let pivot = a.col(i)
689            .iter()
690            .enumerate()
691            .skip(i)
692            .filter(|e| !e.1.is_zero())
693            .max_by(|e, f| wt.cmp_abs(e.1, f.1))?.0;
694
695        // Swap the pivot row with the current row.
696        a.swap_rows(pivot, i);
697        for r in i+1..rank {
698            let (pivot_row, row) = a.get_rows_mut(i, r);
699            let fac = wt.neg(wt.div(row[i].clone(), &pivot_row[i]));
700            for c in i+1..rank {
701                wt.mul_add_assign(&mut row[c], &fac, &pivot_row[c]);
702            }
703            let (b_r, b_i) = b.get_mut_entries(r, i);
704            wt.mul_add_assign(b_r, &fac, b_i);
705        }
706    }
707
708    let mut result = Vector::zero(rank);
709    for i in (0..rank).rev() {
710        let row = a.row(i);
711        let sum = (i+1..rank).fold(
712            b[i].clone(),
713            |acc, j| wt.mul_sub(acc, &row[j], &result[j])
714        );
715
716        result[i] = wt.div(sum, &row[i]);
717    }
718
719    Some(result)
720}

722/// Lattice algorithms are possibly internally working with a different ring.
723/// These rings need to implement this.
724pub trait WorkingType: Field + OrderedRing + SqrtRing {
725    fn eps(&self) -> Self::Element;
726
727    fn infinity(&self) -> Self::Element {
728        panic!("`infinity` is not supported for this type.");
729    }
730
731    fn round(s: Self::Element) -> Self::Element;
732
733    fn add_isize(s: Self::Element, i: isize) -> Self::Element;
734}
735
736/// This trait facilitates the conversion between the ring of the lattice and
737/// the ring used internally.
738pub trait WorkingTypeFor<R: Ring>: WorkingType {
739    #[allow(clippy::wrong_self_convention)]
740    fn from_ring(&self, e: &R::Element, r: &R) -> Self::Element;
741
742    fn to_ring(s: &Self::Element, r: &R) -> R::Element;
743}
744
745macro_rules! float_working_type {
746    ($ring:ident) => {
747        impl WorkingType for $ring {
748            fn eps(&self) -> Self::Element {
749                <$ring as Ring>::Element::EPSILON
750            }
751
752            fn infinity(&self) -> Self::Element {
753                <$ring as Ring>::Element::INFINITY
754            }
755
756            fn round(s: Self::Element) -> Self::Element {
757                s.round()
758            }
759
760            fn add_isize(s: Self::Element, i: isize) -> Self::Element {
761                s + i as Self::Element
762            }
763        }
764    };
765}
766
767float_working_type!(F32);
768float_working_type!(F64);
769
770impl WorkingTypeFor<Z> for F32 {
771    fn from_ring(&self, e: &BigInt, _: &Z) -> Self::Element {
772        e.to_f32().unwrap()
773    }
774
775    fn to_ring(s: &Self::Element, _: &Z) -> <Z as Ring>::Element {
776        BigInt::from_f32(s.round()).unwrap()
777    }
778}
779
780impl WorkingTypeFor<Z> for F64 {
781    fn from_ring(&self, e: &BigInt, _: &Z) -> Self::Element {
782        e.to_f64().unwrap()
783    }
784
785    fn to_ring(s: &Self::Element, _: &Z) -> <Z as Ring>::Element {
786        BigInt::from_f64(s.round()).unwrap()
787    }
788}
789
790impl WorkingType for Q {
791    fn eps(&self) -> Self::Element {
792        Q::zero()
793    }
794
795    fn round(s: Self::Element) -> Self::Element {
796        s.round()
797    }
798
799    fn add_isize(s: Self::Element, i: isize) -> Self::Element {
800        s + BigRational::from_isize(i).unwrap()
801    }
802}
803
804impl WorkingTypeFor<Z> for Q {
805    fn from_ring(&self, e: &BigInt, _: &Z) -> Self::Element {
806        BigRational::from_integer(e.clone())
807    }
808
809    fn to_ring(s: &Self::Element, _: &Z) -> <Z as Ring>::Element {
810        s.round().numer().clone()
811    }
812}
